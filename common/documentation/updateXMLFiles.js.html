const fs = require("fs");
const convert = require('xml-js');
const path = require('path');

/**
 * @param {string} filename - filename to be parsed
 * @return {string} browser
*/
function extractBrowser (filename) {
  const fileSplit = filename.split('/').pop();
  const browser = fileSplit.substr(0, fileSplit.indexOf('_'));
  return browser;
};

/**
 * @param {string} dirPath -Directory Path
 * @param {Object} arrayOfFiles - empty object
 * @returns {array} arrayOfFiles - Array of files
 */
function getAllFiles (dirPath, arrayOfFiles) {
  const files = fs.readdirSync(dirPath);

  arrayOfFiles = arrayOfFiles || [];

  files.forEach(file => {
    if (fs.statSync(`${dirPath  }/${  file}`).isDirectory()) {
      arrayOfFiles = getAllFiles(`${dirPath  }/${  file}`, arrayOfFiles);
    } else if(path.extname(file) === ".xml") {
      arrayOfFiles.push(path.join(dirPath, '/', file));
    }
  });

  return arrayOfFiles;
};

/**
 * @returns {undefined}
 */
function updateXMLFiles () {

  const xmlFiles = getAllFiles(`${process.cwd()}/dist/reports`);
  
  xmlFiles.forEach (file => {
    // read file and convert from xml
    const xmlData = convert.xml2js(fs.readFileSync(file), { compact: true });
    // extract browser from filename
    const browser = extractBrowser(file);

    // replace angle brackets as needed
    xmlData.testsuites.testsuite.testcase.forEach((testcase) => {
      if ('failure' in testcase) {
        testcase.failure._attributes.message = testcase.failure._attributes.message.replace('<',  '&lt;');
        testcase.failure._attributes.message = testcase.failure._attributes.message.replace('>',  '&gt;');
      }
    });

    // add browser property
    xmlData.testsuites.testsuite.properties =  {property: {name: 'browser', value: `${browser}`}};

    // convert to xml and write
    fs.writeFileSync(
      file,
      convert.js2xml(xmlData, { compact: true, spaces: 2 }),
    );
  });
}
