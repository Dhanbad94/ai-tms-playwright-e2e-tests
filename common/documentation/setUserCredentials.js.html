const fs = require('fs');
/**
* @param {string} env current environment
* @param {string} module to find the correct user
* @param {string} user to run the tests with
* @returns {object} the credentials selected.
 */
export function setUserCredentials (env, module, user) {
  const allCredentials = JSON.parse(fs.readFileSync(`${ process.cwd() }/config/logins.json`));
  const product =
    process.platform === 'win32' ? module.split('\\')[0] : module.split('/')[0];

  // search credentials for iUser for the product
  const creds = allCredentials[`${env}`][`${product}`].find(obj => obj.username === `${user}`);

  return creds;
}

/**
* @param {string} env current environment
* @param {string} module to find the correct user
* @param {string} key consumer key
* @returns {object} the credentials selected.
 */
export function getUserCredentialsUsingKey(env, module, key) {
  const allCredentials = JSON.parse(fs.readFileSync('logins.json'));
  const product = module.split('/')[0];

  // search credentials for iUser for the product
  const creds = allCredentials[`${env}`][`${product}`].find(
    (obj) => obj.consumerKey === key,
  );

  return creds;
}
